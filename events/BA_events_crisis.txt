							has_fleet_flag = BA_mid_Aris_king
						}
					}
				}
				found_system = {
					move_to = this
					repeat = {
						while = {
							id = Arismid.system.0004							#检测有无爱丽丝舰队
							any_fleet_in_system = {
								is_owned_by = event_target:BA_Precursors_Aris_country
								NOT = {
									has_fleet_flag = BA_mid_Aris_king
								}
							}
						}
						find_random_fleet = {
							trigger = {
								id = Arismid.fleet.0001								#检测爱丽丝舰队
								is_owned_by = event_target:BA_Precursors_Aris_country
								NOT = {
									has_fleet_flag = BA_mid_Aris_king
								}
							}
							found_fleet = {
