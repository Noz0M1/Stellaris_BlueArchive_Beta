				name = "Shittem.name"

					}

						"blue"
						"blue"
						"null"
						"null"
					}
				}
				effect = {
					set_country_flag = BA_Shittem_country
					save_global_event_target_as = BA_Shittem_country
					BA_Shittem_display = yes
				}
			}
		}
		#设置特殊领袖用种族、复活计数器
		create_species = {
			name = "species_name_BA_leader"
			class = Schale
			portrait = "Schale"
			name_list = "Schale"

			immortal = yes

				trait = trait_BA_BDcase

				set_species_flag = BA_leader_species
				save_global_event_target_as = BA_leader_species
			}
		}
		set_variable = {
			which = BA_leader_respawn_cost
			value = 0
		}
		#设置姬器人种族
		if = {
			limit = {
				NOT = {
					any_galaxy_species = {
						has_species_flag = BA_Robot_flag
					}
				}
			}
			create_species = {
				name = "BA_Robot"
				class = MACHINE
				portrait = "Aris"
				namelist = "ArisName"
				traits = {
					trait = trait_test
					trait = trait_mechanical
					trait = trait_robot_emotion_emulators
					trait = trait_robot_logic_engines
					trait = trait_robot_luxurious
					trait = trait_robot_high_bandwidth
				}
				effect = {
					set_species_flag = BA_Robot_flag
					save_global_event_target_as = BA_Robot_species
				}
			}
		}
		#委托参数
		set_variable = {
			which = BA_commission_count
			value = 0
		}
		set_variable = {
			which = BA_Repeatable_commission_01_counter
			value = 1000
		}
		set_variable = {
			which = BA_Repeatable_commission_02_counter
			value = 1500
		}
		set_variable = {
			which = BA_Repeatable_commission_03_counter
			value = 50
		}
		#学园理念
		set_variable = {
			which = BA_SI_cost
			value = -12
		}
		set_variable = {
			which = BA_SI_diff
			value = 60
		}
		#商店计数
		set_variable = {
			which = BA_Angel24_eligma_counter
			value = 10
		}
		#社团限制
		set_variable = {
			which = BA_Club_count
			value = 0
		}
		set_variable = {
			which = BA_Club_limit
			value = 1
		}
		set_variable = {
			which = BA_Club_check
			value = BA_Club_limit
		}
		#面纱相关
		if = {
			limit = {
				has_valid_civic = civic_BA_Omnia_Vanitas
			}
			set_country_flag = BA_Omnia_Vanitas_under_guise
		}
		#失落帝国相关
		set_variable = {
			which = BA_Precursor_system_info
			value = 1
		}
		#内阁相关
		unlock_council_selection = yes
		set_council_position_to_council = councilor_research_kivotos
		random_owned_leader = {
			limit = {
				leader_class = scientist
				is_councilor = no
				is_ruler = no
			}
			set_council_position = councilor_research_kivotos
		}

			}

country_event = {
	id = BA_Shittem.1000
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		from = {
			has_country_flag = BA_Shittem_country
		}
		NOT = {
			has_country_flag = Shittem_engaged
		}
	}
	immediate = {
		BA_Shittem_get_stockpile = yes
		#如果故障中，打开故障提示
		if = {
			limit = {
				always = no
				#has_country_flag = Shittem_malfunctioned
			}
			country_event = {
				id = BA_Shittem.1900
			}
		}
		else = {
			random_list = {
				#事件提前生产0-24的随机数，通过该变量的值进行文本以及肖像的随机变化
				#肖像绑定值，文本绑定值，以此实现肖像与文本的绑定
				#序号为0的肖像与scripted loc当中的BlueArchivePulanaDesc0绑定，以此类推，Arona窗口同理
				1 = {
					set_variable_to_random_value = {
						which = BA_Arona_serial
						min = 0
						#max = 24
						max = 15
						rounded = yes
					}
					country_event = {
						id = BA_Shittem.1001
					}
				}
				1 = {
					set_variable_to_random_value = {
						which = BA_Pulana_serial
						min = 0
						#max = 24
						max = 13
						rounded = yes
					}
					country_event = {
						id = BA_Shittem.1002
					}
				}
			}
		}
	}
}

				remove_country_flag = BA_Normal_commission_01_allow

		every_owned_pop = {
			limit = {
				is_robot_pop = yes
				NOT = {
					species = {
						is_same_species = event_target:BA_Robot_species
					}
				}
			}
			set_pop_flag = BA_to_be_converted
		}
		while = {
			limit = {
				any_owned_pop = {
					has_pop_flag = BA_to_be_converted
				}
			}
			random_owned_pop = {
				limit = {
					has_pop_flag = BA_to_be_converted
				}
				kill_pop = yes
			}
			change_variable = {
				which = BA_MAC_count
				value = 1
			}
		}
		while = {
			count = BA_MAC_count
			create_pop = {
				species = event_target:BA_Robot_species
			}
		}
	}
