country_event = {

	}
	immediate = {
		export_resource_stockpile_to_variable = {
			resource = sr_pyroxenes
			variable = BA_Angel24_resource_cost
		}

			resource = sr_pyroxenes
			variable = BA_Angel24_resource_rare_cost
		}

			which = BA_Angel24_resource_cost
			value = 0.2
		}

		}
	}
