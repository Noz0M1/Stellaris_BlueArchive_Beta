	is_for_colonizable = yes
	category = deposit_cat_blockers

		category = deposit_blockers
		cost = {
			energy = 500
			minerals = 250
		}
		cost = {
			trigger = {
				planet = {
					is_planet_class = pc_BA_campus_planet
				}
			}
			energy = 500
			minerals = 250
		}
	}
	#清除花费1000能量，500矿
	blocker_swap_types = {
		d_Abydos_school
	}
	planet_modifier = {
		planet_max_districts_add = -1
	}
	#-1区划
	drop_weight = {
		weight = 0
	}
	prerequisites = {
		"tech_BA_institute_of_reconstruction"
	}
	on_cleared = {
		if = {
			limit = {
				planet = {
					is_planet_class = pc_BA_ruined_planet
				}
			}
			hidden_effect = {
				#清除后给予flag并检测是否全部清除
				set_Planet_flag = Abydos_was_fixed_flag_ruined
				# planet_event = {
				# 	id = BA_origin.0281
				# }
				planet_event = {
					id = BA_origin.1302
				}
				#委托
				owner = {
					set_country_flag = BA_Normal_commission_02_allow
				}
			}
		}
		else_if = {
			limit = {
				planet = {
					is_planet_class = pc_BA_campus_planet
				}
			}
			hidden_effect = {
				#清除后给予flag并检测是否全部清除
				set_Planet_flag = Abydos_was_fixed_flag
				owner = {

				}
			}

	}
