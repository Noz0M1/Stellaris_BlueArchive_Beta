#下翻页
button_effect_BA_page_down = {
    allow = {
        always = yes
    }
    potential = {
        from = {
            OR = {
                AND = {
                    check_variable = {
                        which = BA_page_serial
                        value < 2
                    }
                    check_variable = {
                        which = BA_page_class
                        value = 1
                    }
                }
                AND = {
                    check_variable = {
                        which = BA_page_serial
                        value < 2
                    }
                    check_variable = {
                        which = BA_page_class
                        value = 2
                    }
                }
                AND = {
                    check_variable = {
                        which = BA_page_serial
                        value < 3
                    }
                    check_variable = {
                        which = BA_page_class
                        value = 4
                    }
                }
                AND = {
                    check_variable = {
                        which = BA_page_serial
                        value < 2
                    }
                    check_variable = {
                        which = BA_page_class
                        value = 7
                    }
                }
            }
        }
    }
    effect = {
        hidden_effect = {
            from = {
                change_variable = {
                    which = BA_page_serial
                    value = 1
                }
            }
        }
    }
}

#上翻页
button_effect_BA_page_up = {
    allow = {
        always = yes
    }
    potential = {
        from = {
            check_variable = {
                which = BA_page_serial
                value > 1
            }
        }
    }
    effect = {
        hidden_effect = {
            from = {
                subtract_variable = {
                    which = BA_page_serial
                    value = 1
                }
            }
        }
    }
}
